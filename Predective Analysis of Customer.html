# Check for missing values in all columns
print(df.isnull().sum())

# Handle missing values. The approach depends on your data. For simplicity, we'll fill missing values with the median for numerical columns and the mode for categorical ones. Adjust as necessary.
for column in df.select_dtypes(include=['int64', 'float64']).columns:
    df[column] = df[column].fillna(df[column].median())

for column in df.select_dtypes(include=['object']).columns:
    df[column] = df[column].fillna(df[column].mode().iloc[0])

# Convert data types
df['num_passengers'] = df['num_passengers'].astype(int)
df['purchase_lead'] = df['purchase_lead'].astype(int)
df['length_of_stay'] = df['length_of_stay'].astype(int)
df['flight_hour'] = df['flight_hour'].astype(int)
df['flight_duration'] = df['flight_duration'].astype(float)
# Boolean columns
df['wants_extra_baggage'] = df['wants_extra_baggage'].astype(bool)
df['wants_preferred_seat'] = df['wants_preferred_seat'].astype(bool)
df['wants_in_flight_meals'] = df['wants_in_flight_meals'].astype(bool)
df['booking_complete'] = df['booking_complete'].astype(bool)

# Deduplicate data
df = df.drop_duplicates()
